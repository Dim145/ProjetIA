9        /// <summary>
10       /// Méthode générique de l'algo de recuit simulé.
11       /// Simple traduction du pseudo code du cours en c#
12       ///
13       /// Toutes les zone plus spécifique doivent être précisé en paramètre de la méthode.
14       /// </summary>
15       /// <param name="x0">valeur initial</param>
16       /// <param name="conditionArret">doit renvoyer true pour arret, false sinon</param>
17       /// <param name="voisin">renvoie une valeur proche de celle envoyer. Dans le cas où "TYpe" est un objet, l'adresse mémoire ne doit pas être la même</param>
18       /// <param name="f">fonction qui renvoie une valeur en float pour déterminé la valeur à maximisé ou minimisé</param>
19       /// <param name="accept">renvoie true pour utilisé la nouvelle valeur, false sinon</param>
20       /// <param name="decroissance">renvoie la nouvelle température en fonction de l'ancienne</param>
21       /// <typeparam name="TYpe">Le type de l'objet qui est manipulé Kakuro par exemple</typeparam>
22       /// <returns></returns>
23       private static TYpe Recuit<TYpe>(TYpe x0, Func<TYpe, float, float, bool> conditionArret, Func<TYpe, TYpe> voisin, Func<TYpe, float> f, Func<float, float, bool> accept, Func<float, float> decroissance)
24       {
25           var x = x0;
26           var t = Tinit;
27           const float Nt = 100f;
28   
29           while (!conditionArret(x, t, Nt))
30           {
31               for (var m = 0; m < Nt; m++)
32               {
33                   var y = voisin(x);
34                   var dF = f(y) - f(x);
35   
36                   if (accept(dF, t))
37                   {
38                       x = y;
39                   }
40               }
41   
42               t = decroissance(t);
43           }
44   
45           return x;
46       }

55       /// <summary>
56       /// Nom de méthode absolument parfait.
57       ///
58       /// Créer un tableau de la taille voulu dont l'adition des valeur = à la valeur total voulu.
59       /// Les valeur du tableau sont random.
60       /// ex: DecompositionPlusRandom(5, 3) peut renvoyer: [2, 1, 2] ou [1, 3, 1] etc...
61       /// </summary>
62       /// <param name="num">Valeur total voulu</param>
63       /// <param name="subNum">Nombres de cases du tableau</param>
64       /// <returns>Un tableau de taille Y dont la somme des valeurs est égale à X</returns>
65       [SuppressMessage("ReSharper.DPA", "DPA0002: Excessive memory allocations in SOH", MessageId = "type: System.SZGenericArrayEnumerator`1[System.Int32]")]
66       public static int[] DecompositionPlusRandom(int num, int subNum)
67       {
68           if (subNum == 1)
69               return new[]{num};
70   
71           int startNum = num;
72           int startSubNum = subNum;
73           bool isGood = false;
74   
75           var res = new int[subNum];
76   
77           while (!isGood)
78           {
79               num = startNum;
80               subNum = startSubNum;
81               res = new int[subNum];
82   
83               int cumulSum = 0;
84               int subSection = 0;
85   
86               int max_random_number = Math.Min(9, startNum);
87   
88               while (true)
89               {
90                   int random_number = random.Next(1, max_random_number + 1);
91   
92                   res[subSection] = random_number;
93                   cumulSum += random_number;
94                   num -= random_number;
95                   subSection++;
96   
97                   if (cumulSum >= startNum)
98                   {
99                       if (cumulSum > startNum)
100                          res[subSection-1] = cumulSum - startNum - (res.Length - subSection);
101  
102                      for (int i = subSection; i < res.Length; i++)
103                          res[i] = 1;
104                      
105                      break;
106                  }
107  
108                  if (subSection == subNum - 1)
109                  {
110                      random_number = num;
111                      res[subSection] = random_number;
112                      cumulSum += random_number;
113                      break;
114                  }
115              }
116  
117              isGood = res.All(i => i is > 0 and < 10);
118          }
119  
120          //random.Shuffle(res);
121  
122          return res;
123      }

125      /// <summary>
126      /// Plus utilisé, cela permettait de mélangé un tableau.
127      /// Originalement utilisé dans la décomposition
128      /// </summary>
129      /// <param name="rng"></param>
130      /// <param name="array"></param>
131      /// <typeparam name="T"></typeparam>
132      public static void Shuffle<T> (this Random rng, T[] array)
133      {
134          int n = array.Length;
135          while (n > 1) 
136          {
137              int k = rng.Next(n--);
138              
139              (array[n], array[k]) = (array[k], array[n]);
140          }
141      }

143      /// <summary>
144      /// Méthode créer pour éviter de polluer le fichier Program.cs.
145      /// </summary>
146      /// <param name="kakuro"></param>
147      /// <returns></returns>
148      [SuppressMessage("ReSharper.DPA", "DPA0001: Memory allocation issues")]
149      public static Kakuro RecuitKakuro(Kakuro kakuro)
150      {
151          var step = 0;
152          
153          return Recuit(kakuro,
154    (kakuro1, f, arg3) =>
155              {
156                  // Selon la taille du kakuro, il ne sert à rien de s'acharner plus que nécéssaire.
157                  if (f < 0.0001f && step > 250*(kakuro1.NbLig*kakuro1.NbCol))
158                  {
159                      Console.WriteLine($"Fin avant d'avoir trouvé de solution. ({kakuro1.ValuesOfInvalidIndices()})");
160                      return true;
161                  }
162                  
163                  step++;
164                  //Console.WriteLine(kakuro1.ValuesOfInvalidIndices());
165                  return !kakuro1.Contains(0) && kakuro1.IsValid();
166              },
167              oldK =>
168              {
169                  int rLig;
170                  int rCol;
171  
172                  // selection d'une case contenant une valeur
173                  int? tmpValue;
174                  do
175                  {
176                      rLig = random.Next(0, oldK.NbLig);
177                      rCol = random.Next(0, oldK.NbCol);
178  
179                      tmpValue = oldK.GetValue(rLig, rCol);
180                  } while (tmpValue is null);
181  
182                  // récupération de ses indices
183                  var indices = oldK.GetIndiceOfValue(rLig, rCol);
184                  
185                  // clone de la valeur actuelle
186                  var newK = oldK.PseudoClone();
187                  
188                  var indiceLig = indices[1];
189                  
190                  // on calcule une décomposition de la ligne
191                  var decompositionLig = DecompositionPlusRandom(indiceLig.indice![1]!.Value, newK.GetTabLengthForIndice(indiceLig.lig, indiceLig.col, true)!.Value);
192  
193                  // qui est directement ajouter sans vérification
194                  for (int i = 0; i < decompositionLig.Length; i++)
195                      newK.SetValue(indiceLig.lig, indiceLig.col + 1 + i, decompositionLig[i]);
196                  
197                  var indiceCol = newK.GetIndiceOfValue(rLig, rCol - (rCol - indices[0].col))[0];
198  
199                  // parcourt de chacunes des cases de la ligne
200                  // cette méthode est plus optimisé que celle fait de base qui consistait
201                  // à juste remplir la ligne et colonne de la case séléctionné
202                  for (int i = 0; i < decompositionLig.Length; i++)
203                  {
204                      // cacule de la case qui croise la ligne sur le tableau de la colonne
205                      var numLigLock = rLig - indiceCol.lig - 1;
206  
207                      // calcule de la décomposition de la colonne
208                      var decompositionCol = DecompositionPlusRandom(
209                          indiceCol.indice![0]!.Value, 
210                          newK.GetTabLengthForIndice(indiceCol.lig, indiceCol.col, false)!.Value
211                      );
212  
213                      // si les cases ont bien la même valeur, on modifie toute la colonne.
214                      // de base d'autres vérification était faite comme est ce que cela ne casse pas les lignes de la case en cours
215                      // mais cela réduisait drastiquement les performances sans pour autant aider pour les grand plateaux.
216                      if (decompositionCol[numLigLock] == decompositionLig[i])
217                      {
218                          for (int j = 0; j < decompositionCol.Length; j++)
219                              newK.SetValue(indiceCol.lig + 1 + j, indiceCol.col, decompositionCol[j]);
220                      }
221  
222                      // Passage à la case suivante.
223                      if (indiceCol.col + 1 < newK.NbCol && newK.GetValue(rLig, indiceCol.col + 1) != null)
224                          indiceCol = newK.GetIndiceOfValue(rLig, indiceCol.col + 1)[0];
225                  }
226  
227                  return newK;
228              },
229              kakuro1 => kakuro1.Count0Value() + kakuro1.ValuesOfInvalidIndices(), // compte le nombre de case valeur vde et la valeur du plateaux
230              (diff, f) =>
231              {
232                  // exactement la méthode présenté dans le cours, a la différence que le random n'est pas exécuté si
233                  // la valeur a <= 0 (inutile puisque nextDouble renvoi une valeur entre 0 et 1)
234                  if (diff > 0) // nouveau - ancien = 12 - 13 = -1
235                  {
236                      var a = (float) Math.Exp( -diff/f );
237                      if( a <= 0 || random.NextDouble() >= a ) 
238                      {
239                          return false;
240                      }
241                  }
242  
243                  return true;
244              },
245              decroissance: f =>
246              {
247                  // la température diminue proportionellement à sa valeur actuelle.
248                  // ce n'est plus que surement pas la meilleur manière de faire.
249                  var value = f - (f / Tinit) * 2;
250  
251                  return value;
252              });
253      }
